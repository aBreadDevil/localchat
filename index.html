    <!-- Firebase Imports -->

    <script type="module">

        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        import { getAuth, signInAnonymously, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";

        import { getFirestore, collection, query, addDoc, onSnapshot, serverTimestamp, orderBy, limit } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        import { setLogLevel } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";


        // Set Firestore log level to Debug for visibility

        setLogLevel('Debug');

        

        // --- Global Variables (MUST be used for Canvas environment) ---

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';

        const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : null;

        const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;


        let db;

        let auth;

        let userId = '';

        let messagesRef;


        const messagesContainer = document.getElementById('messages-container');

        const messageForm = document.getElementById('message-form');

        const messageInput = document.getElementById('message-input');

        const sendButton = document.getElementById('send-button');

        const userInfo = document.getElementById('user-info');

        const loadingIndicator = document.getElementById('loading-indicator');

        const initialMessage = document.getElementById('initial-message');


        /**

         * Initializes Firebase and authenticates the user.

         */

        async function initializeAndAuthenticate() {

            if (!firebaseConfig) {

                console.error("Firebase configuration is missing.");

                loadingIndicator.innerHTML = '<span class="text-red-500">Error: Firebase Config Missing.</span>';

                return;

            }


            try {

                const app = initializeApp(firebaseConfig);

                auth = getAuth(app);

                db = getFirestore(app);


                // Public chat path: /artifacts/{appId}/public/data/wifi_chat

                const publicDataPath = `artifacts/${appId}/public/data`;

                messagesRef = collection(db, publicDataPath, 'wifi_chat');


                // 1. Authenticate

                if (initialAuthToken) {

                    await signInWithCustomToken(auth, initialAuthToken);

                } else {

                    await signInAnonymously(auth);

                }

                

                userId = auth.currentUser?.uid || crypto.randomUUID();

                userInfo.textContent = `Your User ID (Public Identity): ${userId}`;

                loadingIndicator.classList.add('hidden');

                sendButton.disabled = false;

                messageInput.disabled = false;


                // 2. Start listening for messages

                setupMessageListener();


            } catch (error) {

                console.error("Error during Firebase initialization or authentication:", error);

                loadingIndicator.innerHTML = `<span class="text-red-500">Connection Error: ${error.code}</span>`;

            }

        }


        /**

         * Renders a single message element.

         * @param {Object} message - The message object.

         * @returns {string} - The HTML string for the message.

         */

        function renderMessage(message) {

            const isMe = message.userId === userId;

            const align = isMe ? 'self-end' : 'self-start';

            const bgColor = isMe ? 'bg-indigo-500 text-white' : 'bg-gray-100 text-gray-800';

            const bubbleClasses = `max-w-xs md:max-w-md p-3 rounded-xl shadow-md ${align} ${bgColor}`;

            

            let timeString = '';

            if (message.timestamp && message.timestamp.toDate) {

                timeString = message.timestamp.toDate().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });

            }


            return `

                <div class="flex flex-col ${align}">

                    ${!isMe ? `<div class="text-xs text-gray-500 mb-1 font-semibold">${message.userId}</div>` : ''}

                    <div class="${bubbleClasses}">

                        <p class="whitespace-pre-wrap">${message.text}</p>

                        <span class="block text-right ${isMe ? 'text-indigo-200' : 'text-gray-400'} text-xs mt-1">${timeString}</span>

                    </div>

                </div>

            `;

        }



        /**

         * Sets up the real-time listener for messages.

         */

        function setupMessageListener() {

            // Query: Get last 50 messages, ordered by timestamp

            const q = query(

                messagesRef,

                orderBy('timestamp', 'asc') // Ascending order to show oldest first, new messages auto-scroll

            );


            onSnapshot(q, (snapshot) => {

                let newMessagesHtml = '';

                let hasMessages = false;

                snapshot.forEach((doc) => {

                    const message = doc.data();

                    newMessagesHtml += renderMessage(message);

                    hasMessages = true;

                });


                messagesContainer.innerHTML = newMessagesHtml;

                

                if (hasMessages) {

                    initialMessage.classList.add('hidden');

                } else {

                    initialMessage.classList.remove('hidden');

                }


                // Scroll to the bottom of the chat area

                messagesContainer.scrollTop = messagesContainer.scrollHeight;


            }, (error) => {

                console.error("Error listening to messages:", error);

                const errorMessage = document.createElement('p');

                errorMessage.className = 'text-center text-red-500 italic mt-4';

                errorMessage.textContent = 'Could not load chat history.';

                messagesContainer.prepend(errorMessage);

            });

        }


        /**

         * Sends a new message to Firestore.

         * @param {Event} e - The form submission event.

         */

        async function sendMessage(e) {

            e.preventDefault();

            const text = messageInput.value.trim();


            if (!text || !userId) return;


            const newMessage = {

                userId: userId,

                text: text,

                timestamp: serverTimestamp() // Use server timestamp for reliable ordering

            };


            try {

                await addDoc(messagesRef, newMessage);

                messageInput.value = ''; // Clear input field

            } catch (error) {

                console.error("Error sending message:", error);

                // Simple error display instead of alert()

                const errorDisplay = document.createElement('div');

                errorDisplay.className = 'p-2 bg-red-100 text-red-700 rounded-lg mb-2';

                errorDisplay.textContent = 'Failed to send message.';

                messagesContainer.appendChild(errorDisplay);

            }

        }


        // --- Event Listeners and Initial Load ---

        messageForm.addEventListener('submit', sendMessage);


        window.onload = initializeAndAuthenticate;

    </script>
